 38  omitdirs = ['/old/', '/gnu/', '/gwnc/', '/html/'] 
 39  lststfiles = [] 
 40  # selects .txt files in directory tree except if dir in omitdirs 
 41  for dirpath, dirnames, filenames in os.walk('.'): 
 42      for fname in filenames: 
 43          #if fl.endswith('.py'): 
 44          #    nmfl = fl.partition('.py')[0] 
 45          if fname.endswith('.txt'):       # any text file, doctest or not 
 46              tstfnm = dirpath + '/' + fname 
 47              for omtdir in omitdirs: 
 48                  if omtdir in tstfnm:     # testfile in omitted dir 
 49                      break 
 50                  else: 
 51                      if tstfnm in lststfiles:   # if not repeats filenames 
 52                          pass 
 53                      else: 
 54                          lststfiles += [tstfnm] 
 55  #for fl in lststfiles: 
 56  #    print fl 
 57       
 58  modsfail = [] 
 59  for tstfl in lststfiles: 
 60      print '=== Testing ' + tstfl + '... ' 
 61      nmdir, fname = os.path.split(tstfl) 
 62      os.chdir(nmdir) 
 63      #print '    EN ' + os.getcwd() + '; testing file ' + fname 
 64      #pkg = os.path.basename(mndir) 
 65      #print '    package: ', pkg 
 66      #tstrslt = doctest.testfile(fname, package=pkg) 
 67   
 68      tstrslt = doctest.testfile(fname, verbose=False, report=False) 
 69   
 70      #tstrslt = '' 
 71      os.chdir(curdir) 
 72      #print '    EN ' + os.getcwd() 
 73      attempted, failed = tstrslt.attempted, tstrslt.failed 
 74      if attempted == 0: 
 75          print '   Not a test file.' 
 76          continue 
 77      print '   Attempted: ' + str(attempted) + ', failed: ' + str(failed) 
 78      if failed != 0:          # not all tests passed 
 79          modsfail += [(fname, nmdir)]  
 80          raw_input('   Tests FAILED. Press Enter to continue.') 
 81   
 82  if len(modsfail) > 0: 
 83      print '\nThese modules had errors:' 
 84      for (fname, nmdir) in modsfail: 
 85          print '   ' + fname + ' in ' + nmdir 
 86  else: 
 87      print '\nNo modules had errors.' 
 88  print
